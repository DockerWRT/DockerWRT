# DockerWRT
# Usable openwrt, docker and NAS

# To create firmware:
# make 

# To choose a tag:
# make TAG=v23.05.2

# To create toolchain:
# make toolchain

# To using proxy:
# make PROXY=http://192.168.10.2:1080

# To clean:
# make clean

# To distclean:
# rm -rf openwrt-src

# sample:
# make clean
# make TAG=v23.05.2 PROXY=http://192.168.10.2:1080 toolchain
